js常见发送短信倒计时按钮的代码。

先引用jquery.min.js库。

html简单代码

<input type="button" value="发送验证码" class="btn btn-info">

js核心代码

var countdown = 60; 

$('.btn').click(function(){
    settime($(this));
});

function settime(obj) { //发送验证码倒计时
    if (countdown == 0) {
        obj.attr('disabled',false); // 按钮不可点击
        obj.val("免费获取验证码");
        countdown = 60;
        return; // 倒计时结束，跳出循环
    } else {
        obj.attr('disabled',true); // 释放按钮
        obj.val("重新发送(" + countdown + ")");
        countdown--;
    }
    setTimeout(function() { // 闭包函数
            settime(obj) }
        ,1000)
}
